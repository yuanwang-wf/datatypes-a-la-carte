#+begin_src haskell
data Free f a
  = Var a
  | Op (f (Free f a))


data Or k = Or k k

a :: Free Or Int
a = Op (Or (Var 3) (Var 4))

type Alg f a = f a -> a

-- simailr to foldExpr
eval :: Functor f => (f b -> b) -> (a -> b) -> Free f a -> b
eval _ gen (Var x) = gen x
eval alg gen (Op x) = alg (fmap (eval alg gen) x)

-- identity monad ? or Const
data Stop k = Stop
instance Functor Stop where
  fmap _ Stop = Stop

stop :: Free Stop a -> Maybe a
stop = eval alg gen where
  alg :: Stop (Maybe a) -> Maybe a
  alg Stop = Nothing

  gen :: a -> Maybe a
  gen = Just

b :: Free Stop Int
b = Var 3

c :: Free Stop Int
c = Op Stop

main :: IO ()
main = print $ stop c
#+end_src
